// ─── Animated Score Ring ──────────────────────────────────────────────────────
function ScoreRing({score,grade}){
  const cfg=GRADE_CONFIG[grade]||GRADE_CONFIG.Good;
  const r=54, c=64, circ=2*Math.PI*r;
  const dash=circ*(score/100);
  return <svg width={c*2} height={c*2} style={{filter:`drop-shadow(0 0 18px ${cfg.color}55)`}}>
    <circle cx={c} cy={c} r={r} fill="none" stroke="#e2e8f0" strokeWidth={8}/>
    <circle cx={c} cy={c} r={r} fill="none" stroke={cfg.color} strokeWidth={8}
      strokeDasharray={`${dash} ${circ}`} strokeDashoffset={circ/4}
      strokeLinecap="round" style={{transition:"stroke-dasharray 1.2s ease"}}/>
    <text x={c} y={c-6} textAnchor="middle" fontSize={26} fontWeight={800} fill="#0F2044" fontFamily="Georgia,serif">{score}</text>
    <text x={c} y={c+10} textAnchor="middle" fontSize={11} fill="#64748b" fontFamily="inherit">/100</text>
    <text x={c} y={c+24} textAnchor="middle" fontSize={10} fontWeight={700} fill={cfg.color} fontFamily="inherit">{cfg.icon} {grade}</text>
  </svg>;
}
